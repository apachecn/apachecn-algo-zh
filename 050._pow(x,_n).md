class Solution(object):
    def myPow(self, x, n):
        """
        :type x: float
        :type n: int
        :rtype: float
        """
        def myPosPow(x, n):
        	if n == 0 :
        		return 1
        	v = myPosPow(x, n / 2)
        	if n % 2 == 0:
        		return v * v 
        	else:
        		return x * v * v

        if n < 0 :
        	return 1 / myPosPow(x, -n)
        else:
        	return myPosPow(x, n)
