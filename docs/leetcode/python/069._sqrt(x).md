class Solution(object):
    def mySqrt(self, x):
        """
        :type x: int
        :rtype: int
        """
        i = 0
        while i * i <= x :
        	if i * i == x:
        		return i 
        	elif i * i < x and (i+1) * (i+1) > x:
        		return i
        	i += 1

class Solution(object):
    def mySqrt(self, x):
        """
        :type x: int
        :rtype: int
        """
        res = 1.0
        while abs(res * res - x) > 0.1:
            res = (res + x / res) / 2
        return int(res)
