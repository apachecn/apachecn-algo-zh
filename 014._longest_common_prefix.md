class Solution(object):
    def longestCommonPrefix(self, strs):
        """
        :type strs: List[str]
        :rtype: str
        """
        if strs == []: return ''
        
        shortStr = ''
        shortLen = float('inf')

        for s in strs:
        	if len(s) < shortLen:
        		shortLen = len(s)
        		shortStr = s

        # print shortStr, shortLen

        n = len(strs)
        i = 0
        same = True
        while i < shortLen and same:
            j = 0
            while j < n :
                if strs[j][i] != shortStr[i]:
                    same = False
                    break
                j += 1
            if same:
                i += 1
            else:
                break
        return shortStr[:i]
